⌨️ Argo Rollouts CLI Quick Reference
# Watch rollout status in real-time
kubectl argo rollouts get rollout api-service -n prod --watch

# Check current step and analysis
kubectl argo rollouts get rollout api-service -n prod

# Promote to next step (advance past a pause)
kubectl argo rollouts promote api-service -n prod

# Skip all remaining steps → go straight to 100% (emergency)
kubectl argo rollouts promote api-service -n prod --full

# Abort canary → immediately restore 100% stable traffic
kubectl argo rollouts abort api-service -n prod

# Undo to previous stable revision
kubectl argo rollouts undo api-service -n prod

# Retry a failed rollout after fixing the issue
kubectl argo rollouts retry rollout api-service -n prod

# List all AnalysisRuns for a rollout
kubectl get analysisrun -n prod -l rollout-name=api-service

# Describe a specific AnalysisRun to see metric results
kubectl describe analysisrun <name> -n prod

# Watch via Argo Rollouts Dashboard (port-forward)
kubectl argo rollouts dashboard -n prod
